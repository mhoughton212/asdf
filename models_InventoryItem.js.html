models/InventoryItem.js

/**
 * @class InventoryItem
 * @description JavaScript model for an item in the store's inventory
 */
class InventoryItem {
  /**
   * Creates an instance of InventryItem.
   * @param {number} inventoryItemId - the unique id attached to this inventory item
   * @param {String} name = the name of this inventory item
   * @param {String} description - the description of the inventory item
   * @param {number} quantityInStock - the number of this inventory item that are in stock and available to use
   * @param {String} unit - the type of unit of measurement used for this item (pounds, bottles, pieces, etc.)
   * @param {number} recommendedQuantity - the recommended amount of this item to have in stock
   * @param {String} status - the current status of this Inventory Item (active or inactive)
   */
  constructor(inventoryItemId, name, description, quantityInStock, unit, recommendedQuantity, status = 'active') {
    this.inventoryItemId = inventoryItemId;
    this.name = name;
    this.description = description;
    this.quantityInStock = quantityInStock;
    this.unit = unit;
    this.recommendedQuantity = recommendedQuantity;
    this.status = status;
  }
}

export default InventoryItem;
