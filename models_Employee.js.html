/**
 * @class Employee
 * @description JavaScript model for an employee
 */
class Employee {
  /**
   * Creates an instance of Employee.
   * @param employeeId - the unique id of this employee
   * @param {String} firstName - the first name of this employee
   * @param {String} lastName - the last name of this employee
   * @param {String} role - the role this employee plays (Manager, Cashier, etc.)
   * @param {Date} hireDate - the date/time this employee was hired
   * @param payRate - the hourly pay rate of this employee
   */
    constructor(employeeId, firstName, lastName, role, hireDate, payRate, status) {
      this.employeeId = employeeId;
      this.firstName = firstName;
      this.lastName = lastName;
      this.role = role;
      this.hireDate = hireDate;
      this.payRate = payRate;
      this.status = status;
    }
  }
  
//   module.exports = Employee;
export default Employee;
