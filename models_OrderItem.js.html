// const e = require("express");
/**
 * @class OrderItem
 * @description JavaScript model for an order item
 */
class OrderItem {
  /**
   * Creates an instance of OrderItem.
   * @param {number} orderItemId - the unique id of this order item
   * @param {number} orderId - the id of the order this order item belongs to
   * @param {number} menuItemId - the menu item id of this order item
   * @param {number} comboId - the comboId of the combo structure this item was placed on, if any
   * @param {number} quantity - the amount of this item on the order
   * @param {number} price - the cost of this specific order item
   * @param {String} name - the name of this order item
   * @param {MenuItem} - the menuItems that, for a combo {a type of orderItem}, belong to an item on an order
   */
    constructor(orderItemId, orderId, menuItemId, comboId, quantity, price, name, subItems = []) {
      this.orderItemId = orderItemId;
      this.orderId = orderId;
      this.menuItemId = menuItemId; // Nullable
      this.comboId = comboId;       // Nullable
      this.quantity = quantity;
      this.price = price;
      this.name = name;
      this.subItems = subItems;     // Array of MenuItem (for combos)
    }
  }
  
//   module.exports = OrderItem;
  
  export default OrderItem;
