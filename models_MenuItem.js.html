/**
 * @class MenuItem
 * @description JavaScript model for a menu item on the store's menu
 */
class MenuItem {
  /**
   * Creates an instance of MenuItem.
   * @param {number} menuItemId - the unique id attached to this menu item
   * @param {String} name = the name of this menu item
   * @param {String} description - the description of the menu item
   * @param {number} price - the cost of this menu item
   * @param {String} category - the category of item this is (entree, side, etc.)
   * @param {String} status - whether this is an active or inactive item
   * @param {MenuItemIngredient} ingredients - ingredients that belong to this menu item
   */
    constructor(menuItemId, name, description, price, category, status, ingredients = []) {
      this.menuItemId = menuItemId;
      this.name = name;
      this.description = description;
      this.price = price;
      this.category = category;
      this.status = status;
      this.ingredients = ingredients;
    }
  }
  
    
export default MenuItem;
